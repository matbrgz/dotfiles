# i18n Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add react-i18next internationalization to the GUI app with English, pt-BR, and Spanish, with a language switcher in the Header persisted to localStorage.

**Architecture:** `apps/gui/src/i18n.ts` initializes i18next with 9 namespaces × 3 languages imported as static JSON. A 3-button toggle in `Header.tsx` calls `i18n.changeLanguage()` and saves to `localStorage('dotfiles-lang')`. Each component calls `useTranslation('namespace')` and replaces hardcoded strings with `t('key')`.

**Tech Stack:** react-i18next, i18next, TypeScript, Vite (static JSON imports), React 18

**Branch note:** Create the worktree from `feat/disk-cleaner-redesign` (or master if that branch has already been merged) — the git tab tasks (Task 10) require `GitReposTab.tsx` and `GitRepoDetail.tsx` which live on that branch.

```tsx
import React from 'react';
import { type ProgramManifest } from '@dotfiles/schema';
import { ProgramCard } from './ProgramCard';
import { useTranslation } from 'react-i18next';

interface InventoryTabProps {
  registry: Record<string, ProgramManifest>;
  searchQuery: string;
  installing: Record<string, boolean>;
  onInstall: (id: string) => void;
  installedStatus: Record<string, boolean>;
  scanning: boolean;
}

export const InventoryTab: React.FC<InventoryTabProps> = ({
  registry, searchQuery, installing, onInstall, installedStatus, scanning,
}) => {
  const { t } = useTranslation('inventory');
  const term = searchQuery.toLowerCase();
  const filtered = Object.entries(registry).filter(([id, p]) =>
    !term || id.includes(term) || p.name.toLowerCase().includes(term) || p.description.toLowerCase().includes(term)
  );
  const categories = Array.from(new Set(filtered.map(([, p]) => p.category))).sort();

  if (filtered.length === 0) {
    return (
      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'center', padding: '60px 0' }}>
        <p style={{ color: 'var(--color-text-3)', fontSize: 12 }}>{t('emptySearch', { query: searchQuery })}</p>
      </div>
    );
  }

  return (
    <div style={{ padding: '20px 24px 40px' }}>
      {scanning && (
        <div style={{
          marginBottom: 16, padding: '8px 14px',
          background: 'var(--color-amber-bg)', border: '1px solid rgba(251,191,36,0.2)',
          borderRadius: 6, fontSize: 11, color: 'var(--color-amber)',
        }}>
          {t('scanning')}
        </div>
      )}

      {categories.map(cat => {
        const items = filtered.filter(([, p]) => p.category === cat);
        if (!items.length) return null;
        return (
          <section key={cat} style={{ marginBottom: 32 }}>
            <div style={{ display: 'flex', alignItems: 'center', gap: 12, marginBottom: 12 }}>
              <span style={{ fontSize: 10, fontWeight: 600, color: 'var(--color-text-3)', textTransform: 'uppercase', letterSpacing: '0.1em' }}>
                {cat}
              </span>
              <span style={{ fontSize: 10, color: 'var(--color-text-3)' }}>
                ({items.filter(([id]) => installedStatus[id]).length}/{items.length})
              </span>
              <div style={{ flex: 1, height: 1, background: 'var(--color-border)' }} />
            </div>

            <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fill, minmax(220px, 1fr))', gap: 10 }}>
              {items.map(([id, prog]) => (
                <ProgramCard
                  key={id}
                  title={prog.name}
                  description={prog.description}
                  category={prog.category}
                  onAction={() => onInstall(id)}
                  actionLabel={
                    installing[id] ? t('installing') :
                    installedStatus[id] ? t('reinstall') :
                    scanning ? '...' :
                    t('install')
                  }
                  status={
                    installing[id] ? 'installing' :
                    installedStatus[id] ? 'success' :
                    'ready'
                  }
                />
              ))}
            </div>
          </section>
        );
      })}
    </div>
  );
};
```
